/** * מערכת ניהול מלאי - Apps Script Web App * תומך בפעולות מנהל: עדכון כמות, עדכון תמונה (ישירות ל-ImgBB ולשורה הספציפית בשיטס), הוספה ומחיקה. */ // משיכת המפתחות והסיסמאות בצורה מאובטחת ממשתני הסביבה של גוגל (Script Properties) // בטוח לחלוטין להעלאה ל-GitHub! var scriptProps = PropertiesService.getScriptProperties(); var SECRET = scriptProps.getProperty('ADMIN_SECRET'); var IMGBB_API_KEY = scriptProps.getProperty('IMGBB_API_KEY'); function doGet(e) { return ContentService.createTextOutput("השרת פעיל ומחובר למלאי המחסן!"); } function doPost(e) { var lock = LockService.getScriptLock(); lock.waitLock(15000); try { var params = JSON.parse(e.postData.contents); // חסימת ביצוע אם המשתמש לא הזין סיסמה תקינה או שהסיסמה לא הוגדרה בשרת if (!SECRET || String(params.secret) !== String(SECRET)) { return jsonResponse({ status: "error", message: "סיסמת מנהל שגויה או לא מוגדרת בשרת!" }); } var action = params.action || 'updateQuantity'; var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; if (action === 'updateQuantity') { return handleUpdateQuantity(sheet, params); } else if (action === 'updateImage') { return handleUpdateImage(sheet, params); } else if (action === 'deleteRow') { return handleDeleteRow(sheet, params); } else if (action === 'addItem') { return handleAddItem(sheet, params); } else { return jsonResponse({ status: "error", message: "פעולה לא מוכרת: " + action }); } } catch (err) { return jsonResponse({ status: "error", message: "שגיאה בשרת: " + err.message }); } finally { lock.releaseLock(); } } // --- עדכון כמות לשורה קיימת --- function handleUpdateQuantity(sheet, params) { var row = Number(params.row); if (!row || row < 2) return jsonResponse({ status: "error", message: "מספר שורה לא תקין" }); sheet.getRange(row, 3).setValue(String(params.quantity)); // עמודה C return jsonResponse({ status: "ok" }); } // --- עדכון תמונה לשורה ספציפית (מסנכרן את התמונה מהשטח ישירות לשורה הנכונה) --- function handleUpdateImage(sheet, params) { var row = Number(params.row); if (!row || row < 2) return jsonResponse({ status: "error", message: "מספר שורה לא תקין" }); var imageUrl = ''; if (params.imageFile) { imageUrl = uploadToImgBB(params.imageFile); } else if (params.imageUrl) { imageUrl = params.imageUrl; } sheet.getRange(row, 4).setValue(imageUrl); return jsonResponse({ status: "ok", imageUrl: imageUrl }); } // --- מחיקת שורה --- function handleDeleteRow(sheet, params) { var row = Number(params.row); if (!row || row < 2) return jsonResponse({ status: "error", message: "מספר שורה לא תקין" }); sheet.deleteRow(row); return jsonResponse({ status: "ok" }); } // --- הוספת פריט חדש --- function handleAddItem(sheet, params) { var category = params.category || 'כללי'; var name = params.name || ''; var quantity = params.quantity || '0'; var imageUrl = ''; if (!name) return jsonResponse({ status: "error", message: "חסר שם פריט!" }); if (params.imageFile) { imageUrl = uploadToImgBB(params.imageFile); } else if (params.imageUrl) { imageUrl = params.imageUrl; } sheet.appendRow([category, name, quantity, imageUrl]); return jsonResponse({ status: "ok", imageUrl: imageUrl }); } // ============================================================ // פונקציה חכמה שמעלה תמונות ל-ImgBB דרך השרת של גוגל // ============================================================ function uploadToImgBB(base64String) { if (!IMGBB_API_KEY) throw new Error("מפתח API לא מוגדר בשרת!"); var cleanBase64 = base64String.split(',')[1] || base64String; var payload = { "key": IMGBB_API_KEY, "image": cleanBase64 }; var options = { "method": "post", "payload": payload }; var response = UrlFetchApp.fetch("https://api.imgbb.com/1/upload", options); var data = JSON.parse(response.getContentText()); if (data.success) { return data.data.url; } else { throw new Error("שגיאה בהעלאה לשרת התמונות: " + data.status_code); } } function jsonResponse(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); }
